#include <string>

{

  int rez[5 - 1];

  for (int i = 0; i < n; i++)
  {
    for (int j = 0; j < n; j++)
      cin >> a[i][j];
  }

  cout << "nachalnyu dom";

  for (int i = 0; i < n; i++)
    a[i][h] = 0;
  for (int d = 0;d < n - 1; d++)
  {
    for (int j = 0; j < n; j++)
    {
      if (a[h][j] < min && a[h][j] != 0)
      {
        min = a[h][j];
        st = j;
      }
    }
    rez[d] = st;
    h = st;
    for (int i = 0; i < n; i++)
      a[i][h] = 0;

  }
